// Practical 2: Fibonacci
#include<stdio.h>

int fib(int n){
    if(n<=1) return n;
    return fib(n-1)+fib(n-2);
}

int main(){
    int n;
    scanf("%d",&n);

    int a=0,b=1,c;
    printf("%d %d ",a,b);

    for(int i=2;i<n;i++){
        c=a+b;
        printf("%d ",c);
        a=b; b=c;
    }

    printf("\nRecursive=%d",fib(n));
}
