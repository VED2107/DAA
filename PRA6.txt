// Practical 6: Fractional Knapsack
#include<stdio.h>

struct item{
    int v,w;
    float r;
};

int main(){
    int n,cap;
    scanf("%d",&n);

    struct item a[n];
    for(int i=0;i<n;i++){
        scanf("%d%d",&a[i].v,&a[i].w);
        a[i].r=(float)a[i].v/a[i].w;
    }

    scanf("%d",&cap);

    for(int i=0;i<n-1;i++)
        for(int j=i+1;j<n;j++)
            if(a[i].r<a[j].r){
                struct item t=a[i];
                a[i]=a[j]; a[j]=t;
            }

    float total=0;
    for(int i=0;i<n;i++){
        if(cap>=a[i].w){
            cap-=a[i].w;
            total+=a[i].v;
        }else{
            total+=a[i].r*cap;
            break;
        }
    }

    printf("%.2f",total);
}
